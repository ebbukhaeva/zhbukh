function canCutBрус(D, A) { // Диагональ квадрата let diagonal = A * Math.sqrt(2); // Сравнение диагонали с диаметром бревна if (D > diagonal) { return true; // Можно выпилить } else { return false; // Нельзя выпилить } } // Пример использования let diameter = 20; // Диаметр бревна let side = 10; // Сторона квадрата if (canCutBрус(diameter, side)) { console.log("Можно выпилить квадратный брус."); } else { console.log("Нельзя выпилить квадратный брус."); }
